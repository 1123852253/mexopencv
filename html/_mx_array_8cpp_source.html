119 MxArray::MxArray(const cv::Mat& mat, mxClassID classid, bool transpose)
120 {
121  if (mat.empty())
122  {
123  p_ = mxCreateNumericArray(0, 0, mxDOUBLE_CLASS, mxREAL);
124  if (!p_)
125  mexErrMsgIdAndTxt("mexopencv:error", "Allocation error");
126  return;
127  }
128  cv::Mat input = (mat.dims == 2 && transpose) ? mat.t() : mat;
129  // Create a new mxArray.
130  const int nchannels = input.channels();
131  const int* dims_ = input.size;
132  std::vector<mwSize> d(dims_, dims_ + input.dims);
133  d.push_back(nchannels);
134  classid = (classid == mxUNKNOWN_CLASS)
135  ? ClassIDOf[input.depth()] : classid;
136  std::swap(d[0], d[1]);
137  if (classid == mxLOGICAL_CLASS)
138  {
139  // OpenCV's logical true is any nonzero while matlab's true is 1.
140  cv::compare(input, 0, input, cv::CMP_NE);
141  input.setTo(1, input);
142  p_ = mxCreateLogicalArray(d.size(), &d[0]);
143  }
144  else {
145  p_ = mxCreateNumericArray(d.size(), &d[0], classid, mxREAL);
146  }
147  if (!p_)
148  mexErrMsgIdAndTxt("mexopencv:error", "Allocation error");
149  // Copy each channel.
150  std::vector<cv::Mat> channels;
151  split(input, channels);
152  std::vector<mwSize> si(d.size(), 0); // subscript index.
153  int type = CV_MAKETYPE(DepthOf[classid], 1); // destination type.
154  for (int i = 0; i < nchannels; ++i)
155  {
156  si[si.size() - 1] = i; // last dim is a channel index.
157  void *ptr = reinterpret_cast<void*>(
158  reinterpret_cast<size_t>(mxGetData(p_)) +
159  mxGetElementSize(p_) * subs(si));
160  cv::Mat m(input.dims, dims_, type, ptr);
161  channels[i].convertTo(m, type); // Write to mxArray through m.
162  }
163 }

165 MxArray::MxArray(const cv::SparseMat& mat)
166 {
167  if (mat.dims() != 2)
168  mexErrMsgIdAndTxt("mexopencv:error", "cv::Mat is not 2D");
169  if (mat.type() != CV_32FC1)
170  mexErrMsgIdAndTxt("mexopencv:error", "cv::Mat is not float");
171  // Create a sparse array.
172  int m = mat.size(0), n = mat.size(1), nnz = mat.nzcount();
173  p_ = mxCreateSparse(m, n, nnz, mxREAL);
174  if (!p_)
175  mexErrMsgIdAndTxt("mexopencv:error", "Allocation error");
176  mwIndex *ir = mxGetIr(p_);
177  mwIndex *jc = mxGetJc(p_);
178  if (ir == NULL || jc == NULL)
179  mexErrMsgIdAndTxt("mexopencv:error", "Unknown error");
180  // Sort nodes before we put elems into mxArray.
181  std::vector<const cv::SparseMat::Node*> nodes;
182  nodes.reserve(nnz);
183  for (cv::SparseMatConstIterator it = mat.begin(); it != mat.end(); ++it)
184  nodes.push_back(it.node());
185  std::sort(nodes.begin(), nodes.end(), CompareSparseMatNode());
186  // Copy data.
187  double *pr = mxGetPr(p_);
188  int i = 0;
189  jc[0] = 0;
190  for (std::vector<const cv::SparseMat::Node*>::const_iterator
191  it = nodes.begin(); it != nodes.end(); ++it)
192  {
193  mwIndex row = (*it)->idx[0], col = (*it)->idx[1];
194  ir[i] = row;
195  jc[col+1] = i+1;
196  pr[i] = static_cast<double>(mat.value<float>(*it));
197  ++i;
198  }
199 }

348 cv::Mat MxArray::toMat(int depth, bool transpose) const
349 {
350  // Create cv::Mat object.
351  std::vector<int> d(dims(), dims()+ndims());
352  int ndims = (d.size()>2) ? d.size()-1 : d.size();
353  int nchannels = (d.size()>2) ? *(d.end()-1) : 1;
354  depth = (depth==CV_USRTYPE1) ? DepthOf[classID()] : depth;
355  std::swap(d[0], d[1]);
356  cv::Mat mat(ndims, &d[0], CV_MAKETYPE(depth, nchannels));
357  // Copy each channel.
358  std::vector<cv::Mat> channels(nchannels);
359  std::vector<mwSize> si(d.size(), 0); // subscript index
360  int type = CV_MAKETYPE(DepthOf[classID()], 1); // Source type
361  for (int i = 0; i<nchannels; ++i)
362  {
363  si[d.size()-1] = i;
364  void *pd = reinterpret_cast<void*>(
365  reinterpret_cast<size_t>(mxGetData(p_))+
366  mxGetElementSize(p_)*subs(si));
367  cv::Mat m(ndims, &d[0], type, pd);
368  // Read from mxArray through m
369  m.convertTo(channels[i], CV_MAKETYPE(depth, 1));
370  }
371  cv::merge(channels, mat);
372  return (mat.dims==2 && transpose) ? cv::Mat(mat.t()) : mat;
373 }

390 {
391  // Check if it's sparse.
392  if (!isSparse() || !isDouble())
393  mexErrMsgIdAndTxt("mexopencv:error", "MxArray is not sparse");
394  mwIndex *ir = mxGetIr(p_);
395  mwIndex *jc = mxGetJc(p_);
396  if (ir == NULL || jc == NULL)
397  mexErrMsgIdAndTxt("mexopencv:error", "Unknown error");
398  // Create cv::SparseMat.
399  int m = mxGetM(p_), n = mxGetN(p_);
400  int dims[] = {m, n};
401  cv::SparseMat mat(2, dims, CV_32F);
402  // Copy data.
403  double *pr = mxGetPr(p_);
404  for (int j=0; j<n; ++j)
405  {
406  mwIndex start = jc[j], end = jc[j + 1] - 1;
407  // (row,col) <= val.
408  for (mwIndex i = start; i <= end; ++i)
409  mat.ref<float>(ir[i], j) = static_cast<float>(pr[i]);
410  }
411  return mat;
